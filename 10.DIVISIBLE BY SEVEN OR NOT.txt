             10.DIVISIBLE BY SEVEN OR NOT 

CODE:


#include <stdio.h>

void main() {
    int a,b;
    printf("Enter A value:  ");
    scanf("%d",& a);
    if(a %7==0){
        printf("THE ENTERED VALUE IS DIVISIBLE BY 7(SEVEN) ");
    }
    else{
         printf("THE ENTERED VALUE IS  NOT DIVISIBLE BY 7(SEVEN) ");
    }
}
